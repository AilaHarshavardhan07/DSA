import java.util.*;
import java.lang.*;
import java.io.*;

class Codechef
{
	public static void main (String[] args) throws java.lang.Exception
	{
		Scanner sc = new Scanner(System.in);
		int x = sc.nextInt();
		int rev = 0;
		int temp = x;
		while(temp>0){
		    int rem = temp % 10;
		    rev = rev * 10 + rem;
		    temp = temp/10;
		}
		if(x==rev){
		    System.out.println("The given number is palindrome");
		}
		else{
		    System.out.println("The given number is Not palindrome");
		}
		sc.close();
	}
}
